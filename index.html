// ======================= // SAMAWI AGRO MAIN SERVER // ======================= const express = require("express"); const mongoose = require("mongoose"); const cors = require("cors"); const jwt = require("jsonwebtoken"); const bcrypt = require("bcryptjs"); const app = express(); app.use(express.json()); app.use(cors()); // ======================= // DATABASE CONNECTION // ======================= mongoose.connect("mongodb://127.0.0.1:27017/samawi_agro", { useNewUrlParser: true, useUnifiedTopology: true, }) .then(() => console.log("Database Connected")) .catch(err => console.log(err)); // ======================= // MODELS // ======================= // USER const User = mongoose.model("User", { name: String, phone: String, password: String, role: { type: String, default: "buyer" }, // buyer, seller, admin }); // PRODUCT const Product = mongoose.model("Product", { name: String, price: Number, description: String, image: String, sellerId: String, }); // ORDER const Order = mongoose.model("Order", { buyerId: String, products: Array, total: Number, status: { type: String, default: "pending" }, }); // ======================= // AUTH MIDDLEWARE // ======================= const auth = (req, res, next) => { const token = req.headers["authorization"]; if (!token) return res.status(401).send("No token"); try { const decoded = jwt.verify(token, "SECRET_KEY"); req.user = decoded; next(); } catch { res.status(401).send("Invalid token"); } }; // ======================= // REGISTER // ======================= app.post("/api/register", async (req, res) => { const { name, phone, password } = req.body; const hashed = await bcrypt.hash(password, 10); const user = new User({ name, phone, password: hashed, }); await user.save(); res.send("User created"); }); // ======================= // LOGIN // ======================= app.post("/api/login", async (req, res) => { const { phone, password } = req.body; const user = await User.findOne({ phone }); if (!user) return res.send("User not found"); const valid = await bcrypt.compare(password, user.password); if (!valid) return res.send("Wrong password"); const token = jwt.sign( { id: user._id, role: user.role }, "SECRET_KEY" ); res.json({ token }); }); // ======================= // ADD PRODUCT (ADMIN ONLY) // ======================= app.post("/api/product", auth, async (req, res) => { if (req.user.role !== "admin") return res.send("Access denied"); const product = new Product(req.body); await product.save(); res.send("Product added"); }); // ======================= // GET PRODUCTS // ======================= app.get("/api/products", async (req, res) => { const products = await Product.find(); res.json(products); }); // ======================= // CREATE ORDER // ======================= app.post("/api/order", auth, async (req, res) => { const order = new Order({ buyerId: req.user.id, products: req.body.products, total: req.body.total, }); await order.save(); res.send("Order created"); }); // ======================= // PI PAYMENT (READY) // ======================= app.post("/api/pay/pi", auth, async (req, res) => { // HAPA utaunganisha Pi SDK // Step: // 1. Tuma request kwa Pi Wallet // 2. Subiri confirmation // 3. Update order status res.send("Pi payment initiated (connect SDK here)"); }); // ======================= // ADMIN CHECK // ======================= app.get("/api/admin-check", auth, (req, res) => { if (req.user.role === "admin") { res.send("Welcome Admin"); } else { res.send("Not Admin"); } }); // ======================= // START SERVER // ======================= app.listen(3000, () => { console.log("Server running on port 3000"); });
